9. 피자 조각 개수와 사람 수를 입력받아, 한 사람당 몇 조각 먹을 수 있는지, 남는 조각은 몇 개인지 출력
package homeworkday2;

import java.util.Scanner;

public class Quiz9 {

	public static void main(String[] args) {
		
		 Scanner scanner = new Scanner(System.in);

	        System.out.print("피자 조각 수를 입력하세요: ");
	        int pizzaSlices = scanner.nextInt(); // 피자 조각 수

	        System.out.print("사람 수를 입력하세요: ");
	        int numberOfPeople = scanner.nextInt(); // 사람 수

	        
	        if (numberOfPeople == 0) {
	            System.out.println("사람 수가 0명이므로 나눌 수 없습니다.");
	        } else {
	            int slicesPerPerson = pizzaSlices / numberOfPeople; // 한 사람당 먹을 수 있는 조각 수
	            int remainingSlices = pizzaSlices % numberOfPeople; // 남는 조각 수

	            System.out.println("한 사람당 먹을 수 있는 피자 조각 수: " + slicesPerPerson + "조각");
	            System.out.println("남는 피자 조각 수: " + remainingSlices + "조각");
	        }

	        scanner.close();
	    }
	}

    int dice1 = random.nextInt(6) + 1; // nextInt(6)은 0부터 5까지의 수를 반환하므로 1을 더함
    System.out.println("첫 번째 주사위: " + dice1);

    System.out.println("두 주사위의 합: " + sum);
}
}
11.  1부터 50까지의 랜덤 숫자 생성
package homeworkday2;

import java.util.Random;
import java.util.Scanner;

public class Quiz11 {

	public static void main(String[] args) {
		 
        Random random = new Random();
        int answer = random.nextInt(50) + 1; // 0~49 + 1 -> 1~50
        Scanner scanner = new Scanner(System.in);
        int userGuess;
        boolean correct = false;

        System.out.println("1부터 50 사이의 숫자를 맞춰보세요!");

        
        while (!correct) {
            System.out.print("숫자를 입력하세요: ");
            userGuess = scanner.nextInt();

           
            if (userGuess > answer) {
                System.out.println("DOWN");
            } else if (userGuess < answer) {
                System.out.println("UP");
            } else {
                
                System.out.println("정답입니다! 게임을 종료합니다.");
                correct = true;
            }
        }

        
        scanner.close();
    }
}
